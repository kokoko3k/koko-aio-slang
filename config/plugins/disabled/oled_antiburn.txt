// **Antoburn protection (OLEDs):**
//    . Enabling this disables LCD antighosting.
//
//    It works by slowly shaking the image in order to prevent image retention
//
//    Shake just content (0) or whole screen (1):
//        Shaking whole screen snaps to screen pixels to preserve mask definition, so it may
//        produce a jumpy effect, while content only shaking is smooth and not so noticeable.
//        You may not need whole screen shaking when ambientlight is colorizing  the background pictures
//        or the bezel, or if you don't use bezel or bg images at all, ofc.
//
//    Shake amplitude:
//        Modulates the shakle size.
//
//    Shake speed over X/Y axis
//        Modulates the frequency of the shake in time.


#ifndef DO_ANTIBURN_PLUGIN
#define DO_ANTIBURN_PLUGIN
#endif

#pragma parameter LABEL0162	         "[Plugin] OLED Antiburn:"   0 0 0 1
#pragma parameter ANTIBURN_COMPLETE  "       Shake just content (0) or whole screen (1)"   0.0  0.0 1.0 1.0
#pragma parameter ANTIBURN_AMPLITUDE "       Shake amplitude"   		1.0  0.0 10.0 1.0
#pragma parameter ANTIBURN_X         "       Shake speed over X axis"   1.0  0.0 10.0 1.0
#pragma parameter ANTIBURN_Y         "       Shake speed over Y axis"   1.0  0.0 10.0 1.0
